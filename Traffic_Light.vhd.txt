library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity traffic_light is
    Port (
        clk         : in  STD_LOGIC;
        reset       : in  STD_LOGIC;
        NS_traffic  : in  STD_LOGIC; -- 1 = high traffic
        EW_traffic  : in  STD_LOGIC;

        NS_red      : out STD_LOGIC;
        NS_yellow   : out STD_LOGIC;
        NS_green    : out STD_LOGIC;

        EW_red      : out STD_LOGIC;
        EW_yellow   : out STD_LOGIC;
        EW_green    : out STD_LOGIC
    );
end traffic_light;

architecture Behavioral of traffic_light is

    -- FSM states
    type state_type is (S0, S1, S2, S3);
    signal current_state, next_state : state_type;
    
      -- Clock divider
    signal slow_clk : STD_LOGIC := '0';
    signal clk_count : integer := 0;

    -- Timer
    signal counter : integer := 0;
    signal green_time_NS : integer := 20;
    signal green_time_EW : integer := 20;
    signal yellow_time   : integer := 8;

begin

-- CLOCK DIVIDER (slow down 100MHz clock)
process(clk)
begin
    if rising_edge(clk) then
        clk_count <= clk_count + 1;

        if clk_count = 20000000 then
            slow_clk <= not slow_clk;
            clk_count <= 0;
        end if;

    end if;
end process;

-- 🔁 STATE REGISTER + COUNTER
process(slow_clk, reset)
begin
    if reset = '1' then
        current_state <= S0;
        counter <= 0;

    elsif rising_edge(slow_clk) then
        if current_state /= next_state then
            counter <= 0;
        else
            counter <= counter + 1;
        end if;

        current_state <= next_state;
    end if;
end process;

-- ⚙️ ADAPTIVE LOGIC
process(NS_traffic, EW_traffic)
begin
    if NS_traffic = '1' then
        green_time_NS <= 15;  -- longer green
    else
        green_time_NS <= 5;
    end if;

    if EW_traffic = '1' then
        green_time_EW <= 15;
    else
        green_time_EW <= 5;
    end if;
end process;

-- 🔄 NEXT STATE LOGIC
process(current_state, counter, green_time_NS, green_time_EW)
begin
    case current_state is

        when S0 =>  -- NS Green
            if counter >= green_time_NS then
                next_state <= S1;
            else
                next_state <= S0;
            end if;

        when S1 =>  -- NS Yellow
            if counter >= yellow_time then
                next_state <= S2;
            else
                next_state <= S1;
            end if;

        when S2 =>  -- EW Green
            if counter >= green_time_EW then
                next_state <= S3;
            else
                next_state <= S2;
            end if;

        when S3 =>  -- EW Yellow
            if counter >= yellow_time then
                next_state <= S0;
            else
                next_state <= S3;
            end if;

    end case;
end process;

-- 💡 OUTPUT LOGIC
process(current_state)
begin
    -- Default OFF
    NS_red <= '0'; NS_yellow <= '0'; NS_green <= '0';
    EW_red <= '0'; EW_yellow <= '0'; EW_green <= '0';

    case current_state is

        when S0 =>  -- NS Green
            NS_green <= '1';
            EW_red <= '1';

        when S1 =>  -- NS Yellow
            NS_yellow <= '1';
            EW_red <= '1';

        when S2 =>  -- EW Green
            EW_green <= '1';
            NS_red <= '1';

        when S3 =>  -- EW Yellow
            EW_yellow <= '1';
            NS_red <= '1';

    end case;
end process;

end Behavioral;
